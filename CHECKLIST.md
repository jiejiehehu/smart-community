```bash

  -H "Content-Type: application/json" \

# 测试创建食堂订单

  -H "Content-Type: application/json" \
  -d '{"dish_ids":[1,2],"customer_name":"测试","meal_type":"lunch"}'

# 测试健康档案
curl -X POST http://localhost:8889/api/health-profile \
  -H "Content-Type: application/json" \
  -d '{"resident_name":"测试","gender":"男","height":175,"weight":70}'
```
